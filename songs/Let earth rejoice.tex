\songtitle{Let earth rejoice}
\begin{song}

Let earth rejoice, let heaven be glad. \\
All people of God sing forever His name. (2x)

\begin{chorus}
    Alleluia! (4x) \\
\end{chorus}

Come, sing to the Lord. Com, sing out your praise,\\
Exult with joy, here is the Lord, our God! (2x)

At the sound of the harp, at the sound of the horn,\\
acclaim Him who comes in the name of the Lord! (2x)
\end{song}
