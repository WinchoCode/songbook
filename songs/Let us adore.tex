\songtitle{Let us adore}
\begin{song}

\begin{chorus}
    Let us adore our Lord and God! \\
    May all the praises of our songs \\
    proclaim his name! \\
    Let us adore our Lord and God! \\
    Release the fire of the Spirit from our hearts!
\end{chorus}

Come to Him, come close to Him today, \\
make of yourselves a living sacrifice!

One in heart, in spirit and in truth \\
we bow before the Lord, the Bread of Life!
\end{song}
