\songtitle{From the rising of the sun}
\begin{song}

\begin{chorus}
From the rising of the sun \\
to the going down of the same \\
The Lord's name is to be praised. (2x)
\end{chorus}

Praise ye the Lord, \\
praise Him O ye servant of the Lord. \\
Praise the name of the Lord, \\
blessed be the name of the Lord \\
from this time forth and forever more.
\end{song}
