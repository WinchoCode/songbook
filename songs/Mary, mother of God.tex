\songtitle{Mary, Mother of God}
\begin{song}

\begin{chorus}
    Mary, Mother of God, \\
    Mary, Mother of Christ, \\
    Mary, Mother of all, \\
    Mary, Queen of the world!
\end{chorus}

God has chosen you, \\
O Mary humble servant. \\
Virgin, rejoice in your God!

God became a man \\
in you, O Mother Mary. \\
Virgin, rejoice, you are blest!

Spiritfilled, you lived \\
responding to God's promise. \\
Virgin, rejoice with the saints!

You kept in your heart \\
the word of God most holy. \\
Virgin, rejoice, you believed!
\end{song}
