\songtitle{Come, Spirit of love}
\begin{song}

\begin{chorus}
    Come, Spirit of love, enflame our coldness. \\
    Come, kindle your fire, within our hearts.
\end{chorus}

Come, Spirit of light, touch our souls, \\
strengthen every heart in time of need.

Pour on all our wounds, healing love; \\
soften our hard hearts with gentle joy.

Though we go away, guide our steps; \\
rebuild all our hopes faith renew.
\end{song}
